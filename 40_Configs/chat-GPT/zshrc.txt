refresh_oscp_prompt() {
  local my_ip
  my_ip=$(ip -br -4 a show tun0 2>/dev/null | awk '{print $3}' | cut -d/ -f1)
  [ -z "$my_ip" ] && my_ip=$(ip -br -4 a show eth0 2>/dev/null | awk '{print $3}' | cut -d/ -f1)
  CURRENT_MY_IP="${my_ip:-N/A}"

  if [ -n "$TARGET_IP" ]; then
    if [ -n "$TARGET_NAME" ]; then
      TARGET_STATUS="%F{red}[T: $TARGET_IP ($TARGET_NAME)]%f"
    else
      TARGET_STATUS="%F{red}[T: $TARGET_IP]%f"
    fi
  else
    TARGET_STATUS=""
  fi
}

  local selected_list
  selected_list=$(find "$WORDLISTS" "$SECLISTS" -maxdepth 4 -type f 2>/dev/null | fzf \
    --prompt="Select Wordlist > " \
    --preview="echo '[ PATH: {} ]'; echo '[ SIZE: ' \$(wc -l < {} 2>/dev/null) ' lines ]'; echo '------------------'; head -n 15 {}")

  if [ -n "$selected_list" ]; then
    ln -sfn "$selected_list" "$link_name"

    echo "[*] Exported: ACTIVE_DICT=$ACTIVE_DICT"
  else
    echo "No list selected."
  fi
}

# -----------------------------
# tmux environment sync
# -----------------------------
tmuxsync() {
  [ -z "$TMUX" ] && return 0

  local val
  local session

  session=$(tmux display-message -p '#S' 2>/dev/null)
  [ -z "$session" ] && return 0

  val=$(tmux show-environment -t "$session" TARGET_IP 2>/dev/null | sed -n 's/^TARGET_IP=//p')
  [ -n "$val" ] && export TARGET_IP="$val"

  val=$(tmux show-environment -t "$session" TARGET_NAME 2>/dev/null | sed -n 's/^TARGET_NAME=//p')
  export TARGET_NAME="$val"

  val=$(tmux show-environment -t "$session" TARGET_DIR 2>/dev/null | sed -n 's/^TARGET_DIR=//p')
  [ -n "$val" ] && export TARGET_DIR="$val"

  val=$(tmux show-environment -t "$session" OUT 2>/dev/null | sed -n 's/^OUT=//p')
  [ -n "$val" ] && export OUT="$val"

  val=$(tmux show-environment -t "$session" LOG 2>/dev/null | sed -n 's/^LOG=//p')
  [ -n "$val" ] && export LOG="$val"

  val=$(tmux show-environment -t "$session" ASSETS 2>/dev/null | sed -n 's/^ASSETS=//p')
  [ -n "$val" ] && export ASSETS="$val"

  val=$(tmux show-environment -t "$session" workdir 2>/dev/null | sed -n 's/^workdir=//p')
  [ -n "$val" ] && export workdir="$val"

  val=$(tmux show-environment -t "$session" INIT_SCAN_SOURCE 2>/dev/null | sed -n 's/^INIT_SCAN_SOURCE=//p')
  export INIT_SCAN_SOURCE="$val"

  val=$(tmux show-environment -t "$session" INIT_PORTS 2>/dev/null | sed -n 's/^INIT_PORTS=//p')
  export INIT_PORTS="$val"
}

tmuxsync_precmd() {
  [ -n "$TMUX" ] && tmuxsync
}

# -----------------------------
# scan helper
# -----------------------------
extract_ports_gnmap() {
  local file="$1"

  [ ! -f "$file" ] && return 1

  grep -oE '[0-9]+/open' "$file" \
    | cut -d/ -f1 \
    | sort -n -u \
    | paste -sd, -
}

# -----------------------------
# scaninit (gnmap統一版)
# -----------------------------
scaninit() {
  [ -n "$TMUX" ] && tmuxsync

  if [ -z "$TARGET_IP" ] || [ -z "$OUT" ] || [ -z "$LOG" ]; then
    echo "[!] TARGET_IP / OUT / LOG is empty"
    echo "    Run: target <IP> [NAME]"
    return 1
  fi

  mkdir -p "$OUT" "$LOG"

  echo "[*] TARGET_IP=$TARGET_IP"
  echo "[*] OUT=$OUT"

  local ports=""
  local init_source=""
  local session=""

  # -----------------------------
  # 1. Rustscan
  # -----------------------------
  echo "[*] Rustscan fast scan..."
  rustscan -a "$TARGET_IP" -t 2000 --ulimit 5000 -- -Pn -n -oA "$OUT/nmap_initial"

  ports=$(extract_ports_gnmap "$OUT/nmap_initial.gnmap")

  if [ -n "$ports" ]; then
    init_source="$OUT/nmap_initial"
    echo "[+] Open ports (rustscan): $ports"
  else
    # -----------------------------
    # 2. Fallback: top-ports
    # -----------------------------
    echo "[!] Rustscan unreliable → fallback to nmap top-ports"
    nmap -Pn -n --top-ports 1000 "$TARGET_IP" -oA "$OUT/nmap_top1000"

    ports=$(extract_ports_gnmap "$OUT/nmap_top1000.gnmap")

    if [ -n "$ports" ]; then
      init_source="$OUT/nmap_top1000"
      echo "[+] Open ports (fallback): $ports"
    else
      # -----------------------------
      # 3. Fallback: full
      # -----------------------------
      echo "[!] No ports found → running full scan fallback (slow)"
      nmap -Pn -n -p- "$TARGET_IP" -oA "$OUT/nmap_full_fallback"

      ports=$(extract_ports_gnmap "$OUT/nmap_full_fallback.gnmap")

      if [ -n "$ports" ]; then
        init_source="$OUT/nmap_full_fallback"
        echo "[+] Open ports (full fallback): $ports"
      else
        echo "[!] No open ports found at all"
      fi
    fi
  fi

  # -----------------------------
  # 4. Save state
  # -----------------------------
  export INIT_SCAN_SOURCE="$init_source"
  export INIT_PORTS="$ports"

  if [ -n "$TMUX" ]; then
    session=$(tmux display-message -p '#S' 2>/dev/null)
    if [ -n "$session" ]; then
      tmux set-environment -t "$session" INIT_SCAN_SOURCE "$INIT_SCAN_SOURCE"
      tmux set-environment -t "$session" INIT_PORTS "$INIT_PORTS"
    fi
  fi

  # -----------------------------
  # 5. Service scan
  # -----------------------------
  if [ -n "$ports" ]; then
    echo "[*] Running service scan on: $ports"
    nmap -Pn -n -sC -sV -p "$ports" "$TARGET_IP" -oA "$OUT/nmap_service"
  else
    echo "[!] Skipping service scan (no ports)"
  fi

  # -----------------------------
  # 6. Full scan (BG)
  # -----------------------------
  echo "[*] Starting full port scan in background..."
  nohup nmap -Pn -n -p- "$TARGET_IP" \
    -oA "$OUT/nmap_full" \
    > "$LOG/nmap_full.log" 2>&1 &

  echo "[+] Background full scan started"
}

# -----------------------------
# Buckup UDP Scan 1st
# -----------------------------
udp-fast() {
  if [ -z "$TARGET_IP" ] || [ -z "$OUT" ]; then
    echo "[!] TARGET not set"
    return 1
  fi

  # 価値の高いUDPポートだけ
  local ports="53,67,68,69,123,137,138,161,162,500,1900,4500,5355"

  echo "[*] UDP fast scan (high-value ports)"

  sudo nmap -sU -Pn -n -p $ports \
    --max-retries 1 --max-rate 2000 --host-timeout 60s \
    -oA "$OUT/nmap_udp_fast" "$TARGET_IP"
}

# Buckup UDP Scan 2nd

udp-top() {

    --max-retries 2 --max-rate 1000 \

}

udp-deep() {

}

# -----------------------------
# ports-init
# -----------------------------
ports-init() {
  [ -n "$TMUX" ] && tmuxsync

  if [ -z "$OUT" ]; then
    echo "[!] OUT is empty"
    return 1
  fi

  local base=""
  local file=""
  local output=""

  if [ -n "$INIT_SCAN_SOURCE" ] && [ -f "${INIT_SCAN_SOURCE}.nmap" ]; then
    base="$INIT_SCAN_SOURCE"
  elif [ -f "$OUT/nmap_initial.nmap" ]; then
    base="$OUT/nmap_initial"
  elif [ -f "$OUT/nmap_top1000.nmap" ]; then
    base="$OUT/nmap_top1000"
  elif [ -f "$OUT/nmap_full_fallback.nmap" ]; then
    base="$OUT/nmap_full_fallback"
  else
    echo "[!] No usable initial scan result found"
    echo "    Checked:"
    echo "    - $OUT/nmap_initial.nmap"
    echo "    - $OUT/nmap_top1000.nmap"
    echo "    - $OUT/nmap_full_fallback.nmap"
    return 1
  fi

  file="${base}.nmap"
  output=$(grep -E '^[0-9]+/(tcp|udp)\s+open' "$file" | sed 's/  */ /g')

  if [ -z "$output" ]; then
    echo "[!] No open ports found in: $file"
    return 1
  fi

  echo "#### Initial"
  echo '```text'
  echo "$output"
  echo '```'

  printf '%s\n' "$output" | tee /dev/tty | xclip -selection clipboard > /dev/null
  echo "[+] Source: $file"
  echo "[+] Copied to clipboard"
}

# -----------------------------
# next step from port
# -----------------------------
next() {
  [ -n "$TMUX" ] && tmuxsync

  if [ -z "$INIT_PORTS" ]; then
    echo "[!] INIT_PORTS is empty"
    echo "    Run: scaninit"
    return 1
  fi

  echo "========== NEXT ACTION =========="
  echo "[*] Ports: $INIT_PORTS"
  echo

  if [[ "$INIT_PORTS" == *"80"* || "$INIT_PORTS" == *"443"* || "$INIT_PORTS" == *"8080"* || "$INIT_PORTS" == *"8000"* ]]; then
    echo "[P1] Web"
    echo "  → ffuf / gobuster (dir, vhost)"
    echo "  → login / upload / API"
    [[ "$INIT_PORTS" == *"443"* ]] && echo "  → SSL: cert / subdomain / TLS config"
    echo
  fi

  if [[ "$INIT_PORTS" == *"445"* || "$INIT_PORTS" == *"139"* ]]; then
    echo "[P2] SMB"
    echo "  → enum4linux-ng / smbclient"
    echo "  → anonymous / shares"
    echo
  fi

  if [[ "$INIT_PORTS" == *"389"* || "$INIT_PORTS" == *"636"* || "$INIT_PORTS" == *"88"* ]]; then
    echo "[P2] AD / LDAP / Kerberos"
    echo "  → ldapsearch / kerbrute"
    echo "  → AS-REP / user enum"
    echo
  fi

  [[ "$INIT_PORTS" == *"5985"* || "$INIT_PORTS" == *"5986"* ]] && echo "[P3] WinRM → evil-winrm"
  [[ "$INIT_PORTS" == *"3389"* ]] && echo "[P3] RDP → xfreerdp"
  [[ "$INIT_PORTS" == *"22"* ]] && echo "[P3] SSH → creds / 後回し"

  echo
  echo "-----------------------------"
  echo "[*] Always:"
  echo "  → full scan check later"
  echo "  → vhost / subdomain"
  echo "  → creds reuse"
  echo "================================"
}

# -----------------------------
# next step from service scan
# -----------------------------
nextsvc() {
  [ -n "$TMUX" ] && tmuxsync

  if [ -z "$OUT" ]; then
    echo "[!] OUT is empty"
    return 1
  fi

  local file="$OUT/nmap_service.nmap"
  local open_lines

  if [ ! -f "$file" ]; then
    echo "[!] File not found: $file"
    echo "    Run service scan first"
    return 1
  fi

  open_lines=$(grep -E '^[0-9]+/(tcp|udp)\s+open' "$file")

  echo "======= NEXT ACTION (SERVICE) ======="
  echo "[*] Source: $file"
  echo

  if echo "$open_lines" | grep -Eqi 'http|https'; then
    echo "[WEB]"
    echo "  → title / tech / login / upload / API"
    echo "  → ffuf / gobuster / vhost"

    grep -qi 'apache' "$file" && echo "  → Apache: CGI / misconfig / default files"
    grep -qi 'nginx' "$file" && echo "  → Nginx: proxy / config / hidden paths"
    grep -qi 'iis' "$file" && echo "  → IIS: ASP.NET / shortname / web.config"
    grep -qi 'tomcat' "$file" && echo "  → Tomcat: manager / WAR deploy / default creds"
    grep -qi 'jetty' "$file" && echo "  → Jetty: admin / debug / Java app surface"
    grep -qiE 'https|ssl|tls' "$file" && echo "  → SSL: cert CN/SAN / subdomain / TLS config"
    echo
  fi

  if echo "$open_lines" | grep -qiE 'microsoft-ds|netbios|smb'; then
    echo "[SMB]"
    echo "  → enum4linux-ng / smbclient"
    echo "  → anonymous / shares / users"
    grep -qi 'signing' "$file" && echo "  → check SMB signing"
    grep -qi 'samba' "$file" && echo "  → Samba version / known vulns / writable share"
    echo
  fi

  if echo "$open_lines" | grep -qiE 'ldap|kerberos'; then
    echo "[AD / LDAP / KERBEROS]"
    echo "  → ldapsearch / kerbrute / enum"
    echo "  → users / domain / AS-REP / password policy"
    grep -qi 'kerberos' "$file" && echo "  → check AS-REP roast / user enum"
    grep -qi 'ldap' "$file" && echo "  → check naming contexts / anonymous bind"
    echo
  fi

  if echo "$open_lines" | grep -qi 'winrm'; then
    echo "[WINRM]"
    echo "  → evil-winrm (when creds available)"
    echo
  fi

  if echo "$open_lines" | grep -qi 'ms-wbt-server'; then
    echo "[RDP]"
    echo "  → xfreerdp / creds / NLA"
    echo
  fi

  if echo "$open_lines" | grep -qi 'ssh'; then
    echo "[SSH]"
    echo "  → creds / keys / version"
    echo "  → usually lower priority unless creds exist"
    echo
  fi

  if echo "$open_lines" | grep -qi 'telnet'; then
    echo "[TELNET]"
    echo "  → manual login check first"
    echo "  → default / weak creds"
    echo "  → inspect banner / prompt"
    echo
  fi

  if echo "$open_lines" | grep -qi 'ftp'; then
    echo "[FTP]"
    echo "  → anonymous / ls / get / put"
    echo "  → writable upload check"
    echo
  fi

  if echo "$open_lines" | grep -qi 'smtp'; then
    echo "[SMTP]"
    echo "  → user enum / VRFY / EXPN"
    echo
  fi

  if echo "$open_lines" | grep -qi 'mysql'; then
    echo "[MYSQL]"
    echo "  → creds / anonymous / local file / version"
    echo
  fi

  if echo "$open_lines" | grep -qi 'postgresql'; then
    echo "[POSTGRESQL]"
    echo "  → creds / db enum / role check"
    echo
  fi

  if echo "$open_lines" | grep -qi 'mongodb'; then
    echo "[MONGODB]"
    echo "  → no-auth / db list / dump"
    echo
  fi

  if echo "$open_lines" | grep -qi 'redis'; then
    echo "[REDIS]"
    echo "  → unauth / config / file write"
    echo
  fi

  if echo "$open_lines" | grep -qi 'elasticsearch'; then
    echo "[ELASTICSEARCH]"
    echo "  → unauth API / indices / data leak"
    echo
  fi

  if echo "$open_lines" | grep -qi 'docker'; then
    echo "[DOCKER]"
    echo "  → Docker API exposure / container escape surface"
    echo
  fi

  echo "-------------------------------------"
  echo "[*] Notes:"
  echo "  → update Notes / Next Actions in Obsidian"
  echo "  → compare with full scan later"
  echo "====================================="
}

# -----------------------------
# target screenshot
# -----------------------------
fshot() {
  if [ -z "$ASSETS" ]; then
    echo "[!] ASSETS is empty. Run: target <IP> [NAME]"
    return 1
  fi

  mkdir -p "$ASSETS"

  # 撮影前の最新ファイル
  local before
  before=$(ls -t "$ASSETS" 2>/dev/null | head -n 1)

  # 撮影
  flameshot gui --path "$ASSETS"

  # 撮影後の最新ファイル
  local after
  after=$(ls -t "$ASSETS" 2>/dev/null | head -n 1)

  if [ -z "$after" ] || [ "$before" = "$after" ]; then
    echo "[!] Screenshot not saved"
    return 1
  fi

  local fullpath="$ASSETS/$after"

  # Obsidian用リンク（相対パス）
  local rel="assets/$after"

  # クリップボードへ
  echo "![[${rel}]]" | xclip -selection clipboard

  echo "[+] Saved: $fullpath"
  echo "[+] Copied (Obsidian): ![[${rel}]]"
}

# -----------------------------
# Check Full Scan
# -----------------------------
fullcheck() {
  [ -n "$TMUX" ] && tmuxsync

  if [ -z "$TARGET_IP" ] || [ -z "$OUT" ] || [ -z "$LOG" ]; then
    echo "[!] TARGET_IP / OUT / LOG is empty"
    echo "    Run: target <IP> [NAME]"
    return 1
  fi

  local full_base="$OUT/nmap_full"
  local full_log="$LOG/nmap_full.log"
  local full_nmap="${full_base}.nmap"
  local init_nmap=""
  local delta=""

  if [ -n "$INIT_SCAN_SOURCE" ]; then
    init_nmap="${INIT_SCAN_SOURCE}.nmap"
  fi

  if pgrep -af "nmap .* -p- .*${TARGET_IP}" >/dev/null 2>&1; then
    echo "[*] Full scan is still running"
    echo "[*] Progress:"
    [ -f "$full_log" ] && tail -n 20 "$full_log" || echo "[!] Log file not found: $full_log"
    return 0
  fi

  if [ ! -f "$full_nmap" ]; then
    echo "[!] Full scan result not found: $full_nmap"
    return 1
  fi

  echo "[*] Full scan completed"
  echo
  echo "[+] Open ports (full):"
  grep -E '^[0-9]+/(tcp|udp)\s+open' "$full_nmap"

  if [ -n "$init_nmap" ] && [ -f "$init_nmap" ]; then
    delta=$(
      comm -13 \
        <(grep -E '^[0-9]+/(tcp|udp)\s+open' "$init_nmap" | cut -d/ -f1 | sort -n) \
        <(grep -E '^[0-9]+/(tcp|udp)\s+open' "$full_nmap" | cut -d/ -f1 | sort -n)
    )

    echo
    echo "[+] Delta (found later than initial scan):"
    [ -n "$delta" ] && echo "$delta" || echo "None"
  else
    echo
    echo "[!] Initial scan source not found, skipping delta"
  fi
}
# -----------------------------
# Tmux session for infrastructure
# -----------------------------
infratmux() {
  local session="infra"

  if tmux has-session -t "$session" 2>/dev/null; then
    echo "[*] Attaching: $session"
    tmux attach -t "$session"
    return
  fi

  echo "[*] Creating infra session"

  tmux new-session -d -s "$session"

  # --- ペイン構成 ---
  tmux split-window -h -t "$session"
  tmux split-window -v -t "$session:0.1"

  # pane:
  # 0: 左 (VPN / monitor)
  # 1: 右上 (server)
  # 2: 右下 (transfer)

  # --- 左: VPN ---
  tmux send-keys -t "$session:0.0" "cd ~" C-m
  tmux send-keys -t "$session:0.0" "echo '[VPN / Monitor]'" C-m

  # --- 右上: server ---
  tmux send-keys -t "$session:0.1" "cd ~/Transfer" C-m
  tmux send-keys -t "$session:0.1" "echo '[HTTP / Server]'" C-m

  # --- 右下: transfer ---
  tmux send-keys -t "$session:0.2" "cd ~/Transfer" C-m
  tmux send-keys -t "$session:0.2" "echo '[Payload / Transfer]'" C-m

  tmux select-pane -t "$session:0.0"

  tmux attach -t "$session"
}

# -----------------------------
# target
# -----------------------------
target() {
  local ip="$1"
  local name="$2"
  local vault_base="$HOME/Vault/Target"
  local template="$HOME/Vault/Obsidian/80_Templates/Template-Target.md"

  if [ -z "$ip" ]; then
    export TARGET_IP=""
    export TARGET_NAME=""
    export TARGET_DIR=""
    export workdir=""
    export OUT=""
    export LOG=""
    export ASSETS=""
    export INIT_SCAN_SOURCE=""
    export INIT_PORTS=""
    echo "[!] Targeting has been removed."
    return
  fi

  local target_dir="$vault_base/$ip"
  local note_file="$target_dir/${ip}.md"
  local tag="# TARGET"

  mkdir -p "$target_dir/assets" "$target_dir/result/autorecon" "$target_dir/log"

  ln -sfn "$HOME/Workbench" "$target_dir/Workbench"

  export TARGET_IP="$ip"
  export TARGET_NAME="$name"
  export TARGET_DIR="$target_dir"
  export workdir="$target_dir"
  export OUT="$target_dir/result"
  export LOG="$target_dir/log"
  export ASSETS="$target_dir/assets"
  export INIT_SCAN_SOURCE=""
  export INIT_PORTS=""

  if [ -n "$name" ]; then
    sudo sed -i "\| $name $tag|d; \|^$ip |d" /etc/hosts
    echo "$ip $name $tag" | sudo tee -a /etc/hosts > /dev/null
  fi

  if [ ! -f "$note_file" ]; then
    if [ -f "$template" ]; then
      cp "$template" "$note_file"
      sed -i "s|<% tp.file.title %>|$ip|g" "$note_file"
    else
      echo "[!] Template missing: $template"
      cat > "$note_file" <<EOF
# $ip ${name:+($name)}

## Target Info
- IP: $ip
- Hostname: ${name:-}

---

## Quick Start

\`\`\`zsh
target $ip ${name:-}
scaninit
\`\`\`

---

## Port Scan

Primary: \`scaninit\`

> If scaninit fails:
> - Check \`\$TARGET_IP\`
> - Check \`result/\`
> - Check \`log/\`

---

## Open Ports
-

---

## Notes
-

---

## Next Actions
-
EOF
    fi
  fi

  cd "$target_dir" || return

  echo "----------------------------------------"
  echo "Target Set & Moved to $target_dir"
  echo "IP   : $TARGET_IP"
  echo "FQDN : ${TARGET_NAME:-N/A}"
  echo "----------------------------------------"
}

# -----------------------------
# Tmux session for target
# -----------------------------
targettmux() {
  local ip="$1"
  local name="$2"
  local vault_base="$HOME/Vault/Target"

  if [ -z "$ip" ]; then
    echo "[!] Usage: targettmux <IP> [NAME]"
    return 1
  fi

  local target_dir="$vault_base/$ip"
  local out_dir="$target_dir/result"
  local log_dir="$target_dir/log"
  local assets_dir="$target_dir/assets"

  mkdir -p "$assets_dir" "$target_dir/result/autorecon" "$log_dir"
  ln -sfn "$HOME/Workbench" "$target_dir/Workbench"

  # --- セッション名生成 ---
  local last_octet="${ip##*.}"
  local third_octet
  third_octet=$(echo "$ip" | awk -F. '{print $(NF-1)}')

  local safe_name=""
  if [ -n "$name" ]; then
    safe_name="${name// /_}"
    safe_name="${safe_name//[^a-zA-Z0-9_-]/}"
  fi

  local session
  if [ -n "$safe_name" ]; then
    session="${safe_name}_${last_octet}"
  else
    session="t_${third_octet}_${last_octet}"
  fi

  # --- 既存セッション ---
  if tmux has-session -t "$session" 2>/dev/null; then
    echo "[*] Attaching: $session"
    tmux attach -t "$session"
    return
  fi

  echo "[*] Creating session: $session"

  # --- 新規セッション ---
  tmux new-session -d -s "$session" -c "$target_dir"

  # --- セッション単位で環境変数を保持 ---
  tmux set-environment -t "$session" TARGET_IP "$ip"
  tmux set-environment -t "$session" TARGET_NAME "$name"
  tmux set-environment -t "$session" TARGET_DIR "$target_dir"
  tmux set-environment -t "$session" workdir "$target_dir"
  tmux set-environment -t "$session" OUT "$out_dir"
  tmux set-environment -t "$session" LOG "$log_dir"
  tmux set-environment -t "$session" ASSETS "$assets_dir"
  tmux set-environment -t "$session" INIT_SCAN_SOURCE ""
  tmux set-environment -t "$session" INIT_PORTS ""
  tmux set-environment -t "$session" TARGET_SESSION "1"

  tmux set-option -t "$session" -gq @logging-path "$log_dir"

  # --- ペイン構成 ---
  tmux split-window -h -t "$session"
  tmux split-window -v -t "$session:0.1"

  # pane:
  # 0: 左 (scan)
  # 1: 右上 (enum)
  # 2: 右下 (notes)

  # --- 左: scan ---
  tmux send-keys -t "$session:0.0" "target '$ip' '$name'" C-m

  # --- 右上: enum ---
  tmux send-keys -t "$session:0.1" "export TARGET_IP='$ip' TARGET_NAME='$name' TARGET_DIR='$target_dir' workdir='$target_dir' OUT='$out_dir' LOG='$log_dir' ASSETS='$assets_dir' INIT_SCAN_SOURCE='' INIT_PORTS=''" C-m
  tmux send-keys -t "$session:0.1" "cd '$HOME/Workbench'" C-m

  # --- 右下: notes ---
  tmux send-keys -t "$session:0.2" "export TARGET_IP='$ip' TARGET_NAME='$name' TARGET_DIR='$target_dir' workdir='$target_dir' OUT='$out_dir' LOG='$log_dir' ASSETS='$assets_dir' INIT_SCAN_SOURCE='' INIT_PORTS=''" C-m
  tmux send-keys -t "$session:0.2" "cd '$target_dir'" C-m

  # --- アクティブペイン ---
  tmux select-pane -t "$session:0.0"

  tmux attach -t "$session"
}

# -----------------------------
# target reset
# - /etc/hosts の # TARGET を全削除
# - target系 tmux session を全削除
# - 現在シェルの target 変数をクリア
# -----------------------------
targetreset() {
  local current_target_dir="$TARGET_DIR"
  local sessions
  local s

  # --- hosts クリア ---
  sudo sed -i '/# TARGET/d' /etc/hosts

  # --- target directory配下にいるなら退避 ---
  if [[ -n "$current_target_dir" && "$PWD" == "$current_target_dir"* ]]; then
    cd ~ || return
  fi

  # --- target系 tmux session を削除 ---
  sessions=$(tmux list-sessions -F '#S' 2>/dev/null)

  if [[ -n "$sessions" ]]; then
    while IFS= read -r s; do
      [[ -z "$s" ]] && continue

      if tmux show-environment -t "$s" 2>/dev/null | grep -q '^TARGET_SESSION=1$'; then
        tmux kill-session -t "$s"
        echo "[*] Killed tmux session: $s"
      fi
    done <<< "$sessions"
  fi

  # --- 現在シェルの target 環境をクリア ---
  export TARGET_IP=""
  export TARGET_NAME=""
  export TARGET_DIR=""
  export workdir=""
  export OUT=""
  export LOG=""
  export ASSETS=""
  export INIT_SCAN_SOURCE=""
  export INIT_PORTS=""

  echo "[!] All target contexts cleared."
}
